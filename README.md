// ── CONTRIBUTION SNAKE ──
const canvas = document.getElementById('snake');
const ctx = canvas.getContext('2d');
canvas.width = canvas.offsetWidth;

const CELL = 10;
const COLS = Math.floor(canvas.width / CELL);
const ROWS = Math.floor(canvas.height / CELL);

// generate contribution cells
const cells = [];
for (let r = 0; r < ROWS; r++) {
  for (let c = 0; c < COLS; c++) {
    const rnd = Math.random();
    cells.push({ x: c, y: r, active: rnd > 0.55, eaten: false,
      color: rnd > 0.9 ? '#39d353' : rnd > 0.75 ? '#26a641' : '#006d32' });
  }
}

// snake state
let snake = [{ x: 0, y: 0 }];
let dx = 1, dy = 0;
let targetCells = [...cells.filter(c => c.active)];
let currentTarget = null;

function findNext() {
  const head = snake[0];
  // simple BFS-like: pick closest uneaten active cell
  let best = null, bestDist = Infinity;
  targetCells.forEach(tc => {
    if (tc.eaten) return;
    const d = Math.abs(tc.x - head.x) + Math.abs(tc.y - head.y);
    if (d < bestDist) { bestDist = d; best = tc; }
  });
  return best;
}

function moveSnake() {
  if (!currentTarget) currentTarget = findNext();
  if (!currentTarget) { snake = [{ x: 0, y: 0 }]; targetCells.forEach(t=>t.eaten=false); return; }

  const head = snake[0];
  // simple pathfinding: move towards target
  let nx = head.x, ny = head.y;
  if (Math.abs(currentTarget.x - head.x) > Math.abs(currentTarget.y - head.y)) {
    nx += Math.sign(currentTarget.x - head.x);
  } else {
    ny += Math.sign(currentTarget.y - head.y);
  }
  nx = Math.max(0, Math.min(COLS - 1, nx));
  ny = Math.max(0, Math.min(ROWS - 1, ny));

  snake.unshift({ x: nx, y: ny });

  const ate = currentTarget.x === nx && currentTarget.y === ny;
  if (ate) {
    currentTarget.eaten = true;
    currentTarget = null;
  } else {
    if (snake.length > 6) snake.pop();
  }
}

function drawSnake() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  // bg
  ctx.fillStyle = '#161b22';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  // cells
  cells.forEach(c => {
    if (!c.active || c.eaten) {
      ctx.fillStyle = '#21262d';
    } else {
      ctx.fillStyle = c.color;
    }
    ctx.beginPath();
    ctx.roundRect(c.x * CELL + 1, c.y * CELL + 1, CELL - 2, CELL - 2, 2);
    ctx.fill();
  });

  // snake body
  snake.forEach((seg, i) => {
    const alpha = 1 - i / snake.length * 0.6;
    ctx.fillStyle = `rgba(88,166,255,${alpha})`;
    ctx.beginPath();
    ctx.roundRect(seg.x * CELL + 1, seg.y * CELL + 1, CELL - 2, CELL - 2, 3);
    ctx.fill();
  });

  // snake head glow
  const head = snake[0];
  ctx.shadowBlur = 12;
  ctx.shadowColor = '#58a6ff';
  ctx.fillStyle = '#ffffff';
  ctx.beginPath();
  ctx.roundRect(head.x * CELL + 2, head.y * CELL + 2, CELL - 4, CELL - 4, 2);
  ctx.fill();
  ctx.shadowBlur = 0;
}

function snakeLoop() {
  moveSnake();
  drawSnake();
  setTimeout(snakeLoop, 80);
}
snakeLoop();
